\section{Geodesics}

The geodesic equation in general relativity is
\begin{equation}
\label{eq:geodesic_gr}
    \frac{\text{d}^2x^\rho}{\text{d}\tau^2} + \Gamma^\rho_{\mu\nu} \frac{\text{d}x^\mu}{\text{d}\tau} \frac{\text{d}x^\nu}{\text{d}\tau} = 0
\end{equation}
where $\tau$ is the proper time along the geodesic and $\Gamma^\rho_{\mu\nu}$ are the Christoffel symbols. This equation dictates the path followed by both massive and massless particles, given the spacetime metric and the initial conditions of the particles.


\subsection{The 3+1 formalism}

A helpful way to numerically simulate the trajectories of particles in general relativity is by using the Arnowitt-Deser-Misner (ADM) formalism. In this formalism, the metric is written as
\begin{equation}
    g_{\mu\nu} = 
    \begin{pmatrix}
    -\alpha^2+\beta_k\gamma^{kl}\beta_l & \beta_i\\
    \beta_j & \gamma_{ij}
    \end{pmatrix}
\end{equation}
with $\gamma^{ij}$ defined as the inverse of $\gamma_{ij}$, both of which are symmetric because $g_{\mu\nu}$ is symmetric. We will use the notation $\beta^i \equiv \gamma^{ij} \beta_j$. The inverse metric $g^{\mu\nu}$ must satisfy the simple identity $g^{\mu\nu} g_{\nu\rho} = {\delta^\mu}_\rho$. The inverse metric that satisfies this identity is
\begin{equation}
    g^{\mu\nu} = 
    \begin{pmatrix}
    -1/\alpha^2 & \beta^i/\alpha^2\\
    \beta^j/\alpha^2 & \gamma^{ij}-\beta^i\beta^j/\alpha^2
    \end{pmatrix}.
\end{equation}
This is checked in the following equation:
\begin{equation}
\begin{aligned}
    g^{\mu\nu} g_{\nu\rho} &= \begin{pmatrix}
    -1/\alpha^2 & \beta^j/\alpha^2\\
    \beta^i/\alpha^2 & \gamma^{ij}-\beta^i\beta^j/\alpha^2
    \end{pmatrix}
    \begin{pmatrix}
    -\alpha^2+\beta^l\beta_l & \beta_k\\
    \beta_j & \gamma_{jk}
    \end{pmatrix}\\
    &= \begin{pmatrix}
    1-\beta^l\beta_l/\alpha^2+\beta^j\beta_j/\alpha^2 & -\beta_k/\alpha^2+\beta^j\gamma_{jk}/\alpha^2\\
    -\beta_i+\beta_i\beta^l\beta_l/\alpha^2+\gamma^{ij}\beta_j-\beta^i\beta^j\beta_j/\alpha^2 & \beta^i\beta_k/\alpha^2+\gamma^{ij}\gamma_{jk}-\beta^i\beta^j\gamma_{jk}/\alpha^2
    \end{pmatrix}\\
    &= \begin{pmatrix}
    1 & 0\\
    0 & {\delta^i}_k
    \end{pmatrix}.
\end{aligned}
\end{equation}

The task is to write the geodesic equation in the 3+1 formalism. To this end, let $u^\mu \equiv \frac{\text{d}x^\mu}{\text{d}\tau}$ be the contravariant four-velocity of the particle. We also make use of the notation $t \equiv x^0$. To start, let us derive an expression for $\frac{\text{d}x^i}{\text{d}t}$:
\begin{equation}
\label{eq:dx/dt}
\begin{aligned}
    \frac{\text{d}x^i}{\text{d}t} &= \frac{\text{d}\tau}{\text{d}t} \frac{\text{d}x^i}{\text{d}\tau} = \frac{u^i}{u^0} = g^{i\mu} \frac{u_\mu}{u^0} = g^{i0} \frac{u_0}{u^0} + g^{ij} \frac{u_j}{u^0}\\
    &= \frac{\beta^iu_0}{\alpha^2u^0} + \gamma^{ij}\frac{u_j}{u^0} - \frac{\beta^i\beta^ju_j}{\alpha^2u^0}\\
    &= \gamma^{ij}\frac{u_j}{u^0} + \beta^i\frac{u_0-\beta^ju_j}{\alpha^2}\frac{1}{u^0}\\
    &= \gamma^{ij}\frac{u_j}{u^0} - \beta^i\frac{u^0}{u^0}\\
    &= \gamma^{ij}\frac{u_j}{u^0} - \beta^i.
\end{aligned}
\end{equation}
It is also possible to derive an expression for $u^0$ in function of $u_i$. The proper time along the trajectory of the geodesic adheres to the integral
\begin{equation}
    \tau = \int \sqrt{-g_{\mu\nu} \frac{\text{d}x^\mu}{\text{d}\lambda} \frac{\text{d}x^\nu}{\text{d}\lambda}} \text{d}\lambda
\end{equation}
with $\lambda$ a parameter that parameterizes the geodesic. We can choose $\lambda=t$. Along a lightlike geodesic, the proper time does not increase and thus the left hand side of the equation is equal to a constant. Taking the derivative with respect to $t$, the above expression becomes
\begin{equation}
    \epsilon\frac{\text{d}\tau}{\text{d}t} = \sqrt{-g_{\mu\nu} \frac{\text{d}x^\mu}{\text{d}t} \frac{\text{d}x^\nu}{\text{d}t}}.
\end{equation}
The parameter $\epsilon$ is equal to $1$ for massive particles, which move along timelike curves, and equal to $0$ for massless particles, which move along lightlike curves. Taking the square and using equation \eqref{eq:dx/dt}, this expression further becomes
\begin{equation}
\begin{aligned}
    \epsilon\frac{1}{(u^0)^2} = \epsilon\Big(\frac{\text{d}\tau}{\text{d}t}\Big)^2 &= -g_{00} - 2g_{0i}\frac{\text{d}x^i}{\text{d}t} - g_{ij} \frac{\text{d}x^i}{\text{d}t} \frac{\text{d}x^j}{\text{d}t}\\
    &= \alpha^2 - \beta^i\beta_i - 2\beta_i\Big(\gamma^{ij}\frac{u_j}{u^0} - \beta^i\Big) - \gamma_{ij}\Big(\gamma^{ik}\frac{u_k}{u^0} - \beta^i\Big)\Big(\gamma^{jl}\frac{u_l}{u^0} - \beta^j\Big)\\
    &= \alpha^2 - \beta^i\beta_i - \Big(2\beta^i\frac{u_i}{u^0} - 2\beta^i\beta_i\Big) - \Big(\frac{u_j}{u^0} - \beta_j\Big)\Big(\gamma^{jl}\frac{u_l}{u^0} - \beta^j\Big)\\
    &= \alpha^2 - \beta^i\beta_i - \Big(2\beta^i\frac{u_i}{u^0} - 2\beta^i\beta_i\Big) - \Big(\frac{u_j\gamma^{jl}u_l}{(u^0)^2} - 2\beta^j\frac{u_j}{u^0} + \beta_j\beta^j\Big)\\
    &= \alpha^2 - \frac{u_j\gamma^{jl}u_l}{(u^0)^2}\\
    \Rightarrow (u^0)^2 &= \frac{\epsilon + u_i\gamma^{ij}u_j}{\alpha^2}\\
    \Rightarrow u^0 &= \frac{\sqrt{\epsilon + u_i\gamma^{ij}u_j}}{\alpha}.
\end{aligned}
\end{equation}
Before equation \eqref{eq:geodesic_gr} can be expanded using the ADM formalism, there is a relation between $u_0$ and $u^0$ that can be derived and that will be quite useful:
\begin{equation}
\label{eq:u_0}
\begin{aligned}
    u^0 &= g^{0\mu}u_\mu = -\frac{u_0}{\alpha^2} + \frac{\beta^iu_i}{\alpha^2}\\
    \Rightarrow u_0 &= \beta^iu_i - \alpha^2u^0.
\end{aligned}
\end{equation}
Another useful equation is one relating $u^i$ and $u_i$:
\begin{equation}
\label{eq:u^i}
\begin{aligned}
    u^i &= g^{i\mu}u_\mu = g^{i0}u_0 + g^{ij}u_j = \frac{\beta^i}{\alpha^2}\big(\beta^ju_j - \alpha^2u^0\big) + \Big(\gamma^{ij} - \frac{\beta^i\beta^j}{\alpha^2}\Big)u_j\\
    \Rightarrow u^i &= \gamma^{ij}u_j - \beta^iu^0.
\end{aligned}
\end{equation}
The daunting task of expanding the geodesic equation can now be begun. The geodesic equation can be rewritten in the following way:
\begin{equation}
\label{eq:geodesic_derivation_part_0}
\begin{aligned}
    \frac{\text{d}u^\rho}{\text{d}\tau} &= - \Gamma^\rho_{\mu\nu} u^\mu u^\nu\\
    \Rightarrow u^0 \frac{\text{d}u^\rho}{\text{d}t} &= - (u^0)^2\Gamma^\rho_{00} - 2u^0\Gamma^\rho_{0j} u^j - \Gamma^\rho_{jk} u^ju^k\\
    \Rightarrow g_{i\rho}\frac{\text{d}u^\rho}{\text{d}t} &= - u^0g_{i\rho}\Gamma^\rho_{00} - 2g_{i\rho}\Gamma^\rho_{0j} u^j - \frac{1}{u^0}g_{i\rho}\Gamma^\rho_{jk} u^ju^k\\
    \Rightarrow \frac{\text{d}u_i}{\text{d}t} - u^\rho\frac{\text{d}g_{i\rho}}{\text{d}t} &= - u^0g_{i\rho}\Gamma^\rho_{00} - 2g_{i\rho}\Gamma^\rho_{0j} u^j - \frac{1}{u^0}g_{i\rho}\Gamma^\rho_{jk} u^ju^k \quad\text{(using the product rule of derivatives)}.
\end{aligned}
\end{equation}
The term $u^\rho\frac{\text{d}g_{i\rho}}{\text{d}t}$ on the left-hand side can be expanded by using $\frac{\text{d}g_{i\rho}}{\text{d}t} = \frac{\text{d}x^\mu}{\text{d}t} \frac{\partial g^{i\rho}}{\partial\mu}$. The derivation of the geodesic equation can then be continued:
\begin{equation}
\label{eq:geodesic_derivation_part_1}
\begin{aligned}
    \frac{\text{d}u_i}{\text{d}t} - u^\rho\frac{u^\mu}{u^0}\partial_\mu g_{i\rho} &= - u^0g_{i\rho}\Gamma^\rho_{00} - 2g_{i\rho}\Gamma^\rho_{0j} u^j - \frac{1}{u^0}g_{i\rho}\Gamma^\rho_{jk} u^ju^k\\
    \Rightarrow \frac{\text{d}u_i}{\text{d}t} &= u^0\partial_0g_{i0} + u^j\partial_0g_{ij} + u^j\partial_j g_{i0} + \frac{u^k u^j}{u^0}\partial_k g_{ij}\\
    &\quad - u^0g_{i\rho}\Gamma^\rho_{00} - 2g_{i\rho}\Gamma^\rho_{0j} u^j - \frac{1}{u^0}g_{i\rho}\Gamma^\rho_{jk} u^ju^k\\
    &= u^0\partial_0\beta_i + \big(\gamma^{jk}u_k - \beta^ju^0\big)\partial_0\gamma_{ij} + \big(\gamma^{jk}u_k - \beta^ju^0\big)\partial_j\beta_i\\
    &\quad + \frac{1}{u^0}\big(\gamma^{jl}u_l - \beta^ju^0\big)\big(\gamma^{km}u_m - \beta^ku^0\big)\partial_k\gamma_{ij}\\
    &\quad - u^0g_{i\rho}\Gamma^\rho_{00} - 2\big(\gamma^{jk}u_k - \beta^ju^0\big)g_{i\rho}\Gamma^\rho_{0j}\\
    &\quad - \frac{1}{u^0}\big(\gamma^{jl}u_l - \beta^ju^0\big)\big(\gamma^{km}u_m - \beta^ku^0\big)g_{i\rho}\Gamma^\rho_{jk} \quad\text{(using equation \eqref{eq:u^i})}\\
    &= u^0\big[\partial_0\beta_i - \beta^j\partial_0\gamma_{ij} - \beta^j\partial_j\beta_i + \beta^j\beta^k\partial_k\gamma_{ij} - g_{i\rho}\Gamma^\rho_{00} + 2\beta^jg_{i\rho}\Gamma^\rho_{0j} - \beta^j\beta^kg_{i\rho}\Gamma^\rho_{jk}\big]\\
    &\quad + u_j\big[\gamma^{jk}\partial_0\gamma_{ik} + \gamma^{jk}\partial_k\beta_i - \gamma^{jl}\beta^k\partial_k\gamma_{il} - \gamma^{jl}\beta^k\partial_l\gamma_{ik} - 2\gamma^{jk}g_{i\rho}\Gamma^\rho_{0k} + 2\beta^k\gamma^{jl}g_{i\rho}\Gamma^\rho_{kl}\big]\\
    &\quad + \frac{u_ju_k}{u^0}\big[\gamma^{jl}\gamma^{km}\partial_m\gamma_{il} -\gamma^{jl}\gamma^{km}g_{i\rho}\Gamma^\rho_{lm}\big].
\end{aligned}
\end{equation}
The right-hand side of the last line of this equation consists of three parts. Before expanding these three parts one by one, it is instructive to first look at what $g_{i\rho}\Gamma^\rho_{\mu\nu}$ is. Using the formula for the Christoffel symbols gives the following result:
\begin{equation}
    g_{i\rho}\Gamma^\rho_{\mu\nu} = \frac{1}{2}g_{i\rho}g^{\rho\sigma}\big(\partial_\mu g_{\nu\sigma} + \partial_\nu g_{\mu\sigma} - \partial_\sigma g_{\mu\nu}\big) = \frac{1}{2} \big(\partial_\mu g_{\nu i} + \partial_\nu g_{\mu i} - \partial_i g_{\mu\nu}\big).
\end{equation}
Splitting this equation into the three cases $g_{i\rho}\Gamma^\rho_{00}$, $g_{i\rho}\Gamma^\rho_{0j}$, and $g_{i\rho}\Gamma^\rho_{jk}$ we find that
\begin{equation}
\begin{aligned}
    \rightarrow g_{i\rho}\Gamma^\rho_{00} &= \frac{1}{2} \big(2\partial_0 g_{0i} - \partial_i g_{00}\big) = \partial_0\beta_i + \frac{1}{2}\partial_i(\alpha^2 - \beta^j\beta_j)\\
    \rightarrow g_{i\rho}\Gamma^\rho_{0j} &= \frac{1}{2} \big(\partial_0 g_{ij} + \partial_j g_{0i} - \partial_i g_{0j}\big) = \frac{1}{2}\big(\partial_0\gamma_{ij} + \partial_j\beta_i - \partial_i\beta_j\big)\\
    \rightarrow g_{i\rho}\Gamma^\rho_{jk} &= \frac{1}{2} \big(\partial_j g_{ik} + \partial_k g_{ij} - \partial_i g_{jk}\big) = \frac{1}{2} \big(\partial_j \gamma_{ik} + \partial_k \gamma_{ij} - \partial_i \gamma_{jk}\big).
\end{aligned}
\end{equation}
We now expand the first part of the final line of equation \eqref{eq:geodesic_derivation_part_1} (the terms proportional to $u^0$). In the following derivation, colored terms within the same line that have the same color cancel each other out:
\begin{equation}
\begin{aligned}
    &u^0\big[\partial_0\beta_i - \beta^j\partial_0\gamma_{ij} - \beta^j\partial_j\beta_i + \beta^j\beta^k\partial_k\gamma_{ij} - g_{i\rho}\Gamma^\rho_{00} + 2\beta^jg_{i\rho}\Gamma^\rho_{0j} - \beta^j\beta^kg_{i\rho}\Gamma^\rho_{jk}\big]\\
    &= u^0\Big[\textcolor{blue}{\partial_0\beta_i} - \textcolor{green}{\beta^j\partial_0\gamma_{ij}} - \textcolor{purple}{\beta^j\partial_j\beta_i} + \textcolor{red}{\beta^j\beta^k\partial_k\gamma_{ij}} - \textcolor{blue}{\partial_0\beta_i} - \frac{1}{2}\partial_i(\alpha^2 - \beta^j\beta_j) + \beta^j\big(\textcolor{green}{\partial_0\gamma_{ij}} + \textcolor{purple}{\partial_j\beta_i} - \partial_i\beta_j\big)\\
    &\quad\;\;-\beta^j\beta^k\Big(\textcolor{red}{\partial_j \gamma_{ik}} - \frac{1}{2}\partial_i \gamma_{jk}\Big)\Big]\\
    &= u^0 \big[-\alpha\partial_i\alpha + \frac{1}{2}\beta^j\partial_i\beta_j - \frac{1}{2}\beta_j\partial_i\beta^j - \frac{1}{2}\beta^j\beta^k\partial_i\gamma_{jk}\big]\\
    &= u^0 \big[-\alpha\partial_i\alpha + \textcolor{blue}{\frac{1}{2}\beta^j\gamma_{jk}\partial_i\beta^k} + \textcolor{green}{\frac{1}{2}\beta^j\beta^k\partial_i\gamma_{jk}} - \textcolor{blue}{\frac{1}{2}\beta_j\partial_i\beta^j} - \textcolor{green}{\frac{1}{2}\beta^j\beta^k\partial_i\gamma_{jk}}\big]\\
    &= -\alpha u^0\partial_i\alpha.
\end{aligned}
\end{equation}
The second part of equation \eqref{eq:geodesic_derivation_part_1} (the terms proportional to $u^j$) become
\begin{equation}
\begin{aligned}
    &u_j\big[\gamma^{jk}\partial_0\gamma_{ik} + \gamma^{jk}\partial_k\beta_i - \gamma^{jl}\beta^k\partial_k\gamma_{il} - \gamma^{jl}\beta^k\partial_l\gamma_{ik} - 2\gamma^{jk}g_{i\rho}\Gamma^\rho_{0k} + 2\beta^k\gamma^{jl}g_{i\rho}\Gamma^\rho_{kl}\big]\\
    &= u_j\Big[\textcolor{blue}{\gamma^{jk}\partial_0\gamma_{ik}} + \textcolor{green}{\gamma^{jk}\partial_k\beta_i} - \textcolor{purple}{\gamma^{jl}\beta^k\partial_k\gamma_{il}} - \textcolor{red}{\gamma^{jl}\beta^k\partial_l\gamma_{ik}} - \gamma^{jk}\big(\textcolor{blue}{\partial_0\gamma_{ik}} + \textcolor{green}{\partial_k\beta_i} - \partial_i\beta_k\big) + \beta^k\gamma^{jl}\big(\textcolor{purple}{\partial_k \gamma_{il}} + \textcolor{red}{\partial_l \gamma_{ik}} - \partial_i \gamma_{kl}\big)\Big]\\
    &= u_j \big[\gamma^{jk}\partial_i(\beta^l\gamma_{kl}) - \beta^k\gamma^{jl}\partial_i \gamma_{kl}\big]\\
    &= u_j\partial_i\beta^j.
\end{aligned}
\end{equation}
The final part of equation \eqref{eq:geodesic_derivation_part_1} (the terms proportional to $\frac{1}{u^0}u^ju^k$) is
\begin{equation}
\begin{aligned}
    &\frac{u_ju_k}{u^0}\big[\gamma^{jl}\gamma^{km}\partial_m\gamma_{il} -\gamma^{jl}\gamma^{km}g_{i\rho}\Gamma^\rho_{lm}\big]\\
    &= \frac{u_ju_k}{u^0}\gamma^{jl}\gamma^{km}\Big[\textcolor{blue}{\partial_m\gamma_{il}} - \textcolor{blue}{\partial_l \gamma_{im}} + \frac{1}{2}\partial_i\gamma_{lm}\Big]\\
    &= \frac{u_ju_k}{2u^0}\gamma^{jl}\gamma^{km}\partial_i\gamma_{lm}\\
    &= \frac{u_ju_k}{2u^0}\gamma^{jl} \big[\partial_i(\gamma^{km}\gamma_{lm}) - \gamma_{lm}\partial_i\gamma^{km}\big]\\
    &= -\frac{u_ju_k}{2u^0}\partial_i\gamma^{jk} \quad\text{(using $\partial_i{\delta^k}_l = 0$)}.
\end{aligned}
\end{equation}
((The details of the last step need to be resolved.)) The final result of this subsection is achieved by putting the three pieces back together. The geodesic equation in the ADM formalism is thus
\begin{equation}
\label{eq:geodesic_3+1}
    \frac{\text{d}u_i}{\text{d}t} = -\alpha u^0\partial_i\alpha + u_j\partial_i\beta^j - \frac{u_ju_k}{2u^0}\partial_i\gamma^{jk}.
\end{equation}


\subsection{The Electromagnetic force}

When adding the electromagnetic force, the geodesic equation becomes
\begin{equation}
\label{eq:geodesic_gr_lorentz}
    \frac{\text{d}^2x^\rho}{\text{d}\tau^2} + \Gamma^\rho_{\mu\nu} \frac{\text{d}x^\mu}{\text{d}\tau} \frac{\text{d}x^\nu}{\text{d}\tau} = \frac{q}{m}g_{\mu\nu}F^{\rho\mu}\frac{\text{d}x^\nu}{\text{d}\tau}
\end{equation}
where $F^{\mu\nu}$ is the electromagnetic field tensor. Building on the work done in the previous subsection, this geodesic equation can be written in the ADM formalism:
\begin{equation}
\begin{aligned}
    u^0\frac{\text{d}u^\rho}{\text{d}t} &= -\Gamma^\rho_{\mu\nu} u^\mu u^\nu + \frac{q}{m}g_{\mu\nu}F^{\rho\mu}u^\nu\\
    \Rightarrow g_{i\rho}\frac{\text{d}u^\rho}{\text{d}t} &= -g_{i\rho}\Gamma^\rho_{\mu\nu} \frac{u^\mu u^\nu}{u^0} + \frac{q}{m}g_{i\rho}g_{\mu\nu}F^{\rho\mu} \frac{u^\nu}{u^0}\\
    \Rightarrow \frac{\text{d}u_i}{\text{d}t} &= -\alpha u^0\partial_i\alpha + u_j\partial_i\beta^j - \frac{u_j u_k}{2u^0}\partial_i\gamma^{jk} + \frac{q}{m}F_{i\mu}\frac{u^\mu}{u^0}.
\end{aligned}
\end{equation}
The electromagnetic field tensor might not always be easy to obtain, in which case it might be better to write the geodesic equation in terms of the electric and magnetic field vectors. To do this, we use the following definitions:
\begin{equation}
\begin{aligned}
    D^i &= \alpha F^{0i}\\
    H_i &= \frac{1}{2}\alpha \sqrt{\gamma}\,\varepsilon_{ijk}F^{jk}\\
    E_i &= F_{i0}\\
    B^i &= \frac{1}{2}\sqrt{\gamma}^{-1}\varepsilon^{ijk}F_{jk}.
\end{aligned}
\end{equation}
Here $\gamma$ is the determinant of $\gamma_{ij}$ and $\varepsilon_{ijk}$ is the Levi-Civita symbol. Multiplying by $\varepsilon_{ilm}$ and using the property $\varepsilon_{ilm}\varepsilon^{ijk} = {\delta_l}^j {\delta_m}^k - {\delta_l}^k {\delta_m}^j$, the defining relation of $B^i$ becomes
\begin{equation}
\begin{aligned}
    2\sqrt{\gamma}\,\varepsilon_{ilm}B^i &= \big({\delta_l}^j {\delta_m}^k - {\delta_l}^k {\delta_m}^j\big) F_{jk}\\
    \Rightarrow \sqrt{\gamma}\,\varepsilon_{ijk}B^i &= F_{jk} \quad\text{(using $F_{\mu\nu} = -F_{\nu\mu}$)}.
\end{aligned}
\end{equation}
The electromagnetic term in the geodesic equation can then be written as
\begin{equation}
\begin{aligned}
    \frac{q}{m}F_{i\mu}\frac{u^\mu}{u^0} &= \frac{q}{m}F_{i0} + \frac{q}{m}F_{ij}\frac{u^j}{u^0}\\
    &= \frac{q}{m}F_{i0} - \frac{q}{m}F_{ij}\beta^j + \frac{q}{m}F_{ij}\frac{\gamma^{jl}u_l}{u^0} \quad\text{(using equation \eqref{eq:u^i})}\\
    &= \frac{q}{m}F_{i0} - \frac{q}{m}F_{ij}\beta^j + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k\\
    &= -\frac{q}{m}\alpha^2\big(F_{i0}g^{00} + F_{ij}g^{j0}\big) + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k\\
    &= -\frac{q}{m}\alpha^2{F_i}^0 + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k\\
    &= -\frac{q}{m}\alpha^2\big(g_{i0}F^{00} + g_{ij}F^{j0}\big) + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k\\
    &= \frac{q}{m}\alpha^2g_{ij}F^{0j} + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k \quad \text{(using $F^{00} = 0$ and $F^{j0} = -F^{0j}$)}\\
    &= \frac{q}{m}\alpha\gamma_{ij}D^j + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k
\end{aligned}
\end{equation}
The geodesic equation in the ADM formalism in terms of $D^i$ and $B^i$ is thus
\begin{equation}
    \frac{\text{d}u_i}{\text{d}t} = -\alpha u^0\partial_i\alpha + u_j\partial_i\beta^j - \frac{u_j u_k}{2u^0}\partial_i\gamma^{jk} + \frac{q}{m}\alpha\gamma_{ij}D^j + \frac{q}{m}\sqrt{\gamma}\,\varepsilon_{ijk}\frac{\gamma^{jl}u_l}{u^0}B^k.
\end{equation}


\subsection{The Kerr metric in spherical coordinates}

The Kerr metric describes a rotating black hole in general relativity. In spherical coordinates, the metric can be written as
\begin{equation}
    \text{d}s^2 = -\Big(1 - \frac{2Mr}{\rho^2}\Big)\text{d}t^2 - \frac{4Mra\sin^2\theta}{\rho^2}\text{d}\phi\text{d}t + \frac{\rho^2}{\Delta}\text{d}r^2 + \rho^2\text{d}\theta^2 + \Big(r^2 + a^2 + \frac{2Mra^2\sin^2\theta}{\rho^2}\Big)\sin^2\theta\,\text{d}\phi^2
\end{equation}
with $\;\rho^2 \equiv r^2 + a^2 \cos^2\theta\;$ and $\;\Delta \equiv r^2 - 2Mr + a^2$. We see that, in spherical coordinates, the spatial part of the metric is diagonal:
\begin{equation}
    \gamma_{ij} = \begin{pmatrix}
    \gamma_{rr} & \gamma_{r\theta} & \gamma_{r\phi}\\
    \gamma_{\theta r} & \gamma_{\theta\theta} & \gamma_{\theta\phi}\\
    \gamma_{\phi r} & \gamma_{\phi\theta} & \gamma_{\phi\phi}\\
    \end{pmatrix} = \begin{pmatrix}
    \frac{\rho^2}{\Delta} & 0 & 0\\
    0 & \rho^2 & 0\\
    0 & 0 & \Big(r^2 + a^2 + \frac{2Mra^2\sin^2\theta}{\rho^2}\Big)\sin^2\theta
    \end{pmatrix}.
\end{equation}
The metric $\gamma^{ij}$ with the indices up is obtained by taking the inverse, which is easy to do since we have chosen coordinates in which the metric is diagonal:
\begin{equation}
    \gamma^{ij} = \begin{pmatrix}
    \frac{\Delta}{\rho^2} & 0 & 0\\
    0 & \frac{1}{\rho^2} & 0\\
    0 & 0 & \frac{\rho^2}{\big(\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta\big)\sin^2\theta}
    \end{pmatrix}.
\end{equation}
We can also see that $\beta_i$ is
\begin{equation}
    \beta_i = \begin{pmatrix}
    \beta_r\\
    \beta_\theta\\
    \beta_\phi
    \end{pmatrix} = \begin{pmatrix}
    0\\
    0\\
    -\frac{2Mra\sin^2\theta}{\rho^2}
    \end{pmatrix}.
\end{equation}
$\beta^i$ with an upper index can be obtained by multiplying $\beta_j$ with $\gamma^{ij}$:
\begin{equation}
    \beta^i = \begin{pmatrix}
    0\\
    0\\
    \beta_\phi \gamma^{\phi\phi}
    \end{pmatrix} = \begin{pmatrix}
    0\\
    0\\
    -\frac{2Mra}{\sigma^2}
    \end{pmatrix}.
\end{equation}
Here we introduce the notation $\sigma^2 \equiv \rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta$. Finding $\alpha^2$ for the Kerr metric requires a bit more work. Comparing the Kerr metric to the decomposition of the metric in the ADM formalism, we find that
\begin{equation}
\begin{aligned}
    -\alpha^2 + \beta_i\gamma^{ij}\beta_j &= -\Big(1 - \frac{2Mr}{\rho^2}\Big)\\
    \Rightarrow \alpha^2 &= \Big(1 - \frac{2Mr}{\rho^2}\Big) + \beta_\phi^2\gamma^{\phi\phi}\\
    &= \Big(1 - \frac{2Mr}{\rho^2}\Big) + \frac{4M^2r^2a^2\sin^4\theta}{\rho^4 \Big(r^2 + a^2 + \frac{2Mra^2\sin^2\theta}{\rho^2}\Big)\sin^2\theta}\\
    &= \Big(1 - \frac{2Mr}{\rho^2}\Big) + \frac{4M^2r^2a^2\sin^2\theta}{\rho^2\Big(\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta\Big)}\\
    &= \frac{\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta - 2Mr\big(r^2 + a^2\big)}{\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta}\\
    &= \frac{\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta - 2Mr\big(\rho^2 + a^2\sin^2\theta\big)}{\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta} \quad\text{(using $\rho^2 = r^2 + a^2\cos^2\theta$)}\\
    &= \frac{\rho^2\Delta}{\sigma^2}.
\end{aligned}
\end{equation}
To simulate the particle trajectories, we also need the spatial derivatives of $\alpha^2$, $\beta^i$, and $\gamma^{ij}$. The derivative $\partial_r\alpha^2$ is
\begin{equation}
\begin{aligned}
    \partial_r\alpha^2 &= \frac{\Delta}{\sigma^2}\partial_r\rho^2 + \frac{\rho^2}{\sigma^2}\partial_r\Delta - \frac{\rho^2\Delta}{\sigma^4}\partial_r\sigma^2\\
    &= \frac{\Delta}{\sigma^2}(2r) + \frac{\rho^2}{\sigma^2}(2r-2M) - \frac{\rho^2\Delta}{\sigma^4}\big(2r\rho^2 + 2r^3 + 2ra^2 + 2Ma^2\sin^2\theta\big)\\
    &= \frac{\Delta}{\sigma^2}(2r) + \frac{\rho^2}{\sigma^2}(2r-2M) - \frac{\rho^2\Delta}{\sigma^4}\big(4r^3 + 2ra^2\cos^2\theta + 2ra^2 + 2Ma^2\sin^2\theta\big)\\
    &= \frac{2}{\sigma^4}\Big[\Delta\sigma^2r + \rho^2\sigma^2(r-M) - \rho^2\Delta\big(2r^3 + ra^2\cos^2\theta + ra^2 + Ma^2\sin^2\theta\big)\Big]\\
    &= \frac{2}{\sigma^4}\Big[\big(\rho^2r^2 + \textcolor{blue}{\rho^2a^2} + 2Mra^2\sin^2\theta\big)\Delta r + \rho^2\sigma^2(r-M) - \rho^2\Delta\big(2r^3 + ra^2\cos^2\theta + \textcolor{blue}{ra^2} + Ma^2\sin^2\theta\big)\Big]\\
    &= \frac{2}{\sigma^4}\Big[\big(-\rho^2r^3 - \rho^2ra^2\cos^2\theta + Mr^2a^2\sin^2\theta - Ma^4\sin^2\theta\cos^2\theta\big)\Delta + \rho^2\sigma^2(r-M)\Big]\\
    &= \frac{2}{\sigma^4}\Big[\big(-\rho^2r^3 - \rho^2ra^2\cos^2\theta + Mr^2a^2\sin^2\theta - Ma^4\sin^2\theta\cos^2\theta\big)\big(r^2 - 2Mr + a^2\big)\\
    &\qquad + \rho^2\big(\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta\big)(r-M)\Big]\\
    &= \frac{2}{\sigma^4}\Big[\big(-\textcolor{purple}{r^5 - r^3a^2\cos^2\theta} - \textcolor{blue}{r^3a^2\cos^2\theta - ra^4\cos^4\theta} + Mr^2a^2\sin^2\theta - Ma^4\sin^2\theta\cos^2\theta\big)r^2\\
    &\qquad - \big(-r^5 - \textcolor{yellow}{r^3a^2\cos^2\theta} - r^3a^2\cos^2\theta - ra^4\cos^4\theta + Mr^2a^2\sin^2\theta - \textcolor{cyan}{Ma^4\sin^2\theta\cos^2\theta}\big)2Mr\\
    &\qquad + \big(-\textcolor{green}{r^5 - r^3a^2\cos^2\theta} - \textcolor{red}{r^3a^2\cos^2\theta - ra^4\cos^4\theta} + Mr^2a^2\sin^2\theta - Ma^4\sin^2\theta\cos^2\theta\big)a^2\\
    &\qquad + \big(\textcolor{purple}{r^4 + r^2a^2\cos^2\theta} + \textcolor{green}{r^2a^2 + a^4\cos^2\theta} + 2Mra^2\sin^2\theta\big)r^3\\
    &\qquad + \big(\textcolor{blue}{r^4 + r^2a^2\cos^2\theta} + \textcolor{red}{r^2a^2 + a^4\cos^2\theta} + 2Mra^2\sin^2\theta\big)ra^2\cos^2\theta\\
    &\qquad - \big(r^4 + \textcolor{yellow}{r^2a^2\cos^2\theta} + r^2a^2 + a^4\cos^2\theta + 2Mra^2\sin^2\theta\big)Mr^2\\
    &\qquad - \big(\textcolor{yellow}{r^4} + r^2a^2\cos^2\theta + r^2a^2 + a^4\cos^2\theta + \textcolor{cyan}{2Mra^2\sin^2\theta}\big)Ma^2\cos^2\theta\Big]\\
    &= \frac{2M}{\sigma^4}\Big[r^6 + r^4a^2(2\sin^2\theta + \cos^2\theta) - 4Mr^3a^2\sin^2\theta + r^2a^4(\sin^2\theta - \cos^2\theta) - a^6\cos^2\theta\Big].
\end{aligned}
\end{equation}
The derivative $\partial_\theta\alpha^2$ is
\begin{equation}
\begin{aligned}
    \partial_\theta\alpha^2 &= \frac{\Delta}{\sigma^2}\partial_\theta\rho^2 - \frac{\rho^2\Delta}{\sigma^4}\partial_\theta\sigma^2\\
    &= \frac{\Delta}{\sigma^4}\big[\sigma^2\partial_\theta\rho^2 - \rho^2\partial_\theta\sigma^2\big]\\
    &= \frac{\Delta}{\sigma^4}\Big[-2a^2\sin\theta\cos\theta\big(\textcolor{purple}{r^4} + \textcolor{blue}{r^2a^2\cos^2\theta} + \textcolor{cyan}{r^2a^2} + \textcolor{green}{a^4\cos^2\theta} + 2Mra^2\sin^2\theta\big)\\
    &\qquad - r^2\big(-\textcolor{purple}{2r^2a^2\sin\theta\cos\theta} - \textcolor{cyan}{2a^4\sin\theta\cos\theta} + 4Mra^2\sin\theta\cos\theta\big)\\
    &\qquad - a^2\cos^2\theta\big(-\textcolor{blue}{2r^2a^2\sin\theta\cos\theta} - \textcolor{green}{2a^4\sin\theta\cos\theta} + 4Mra^2\sin\theta\cos\theta\big)\Big]\\
    &= \frac{\Delta}{\sigma^4}\Big[-4Mra^4\sin\theta\cos\theta - 4Mr^3a^2\sin\theta\cos\theta\Big] = -4Mra^2(r^2 + a^2)\sin\theta\cos\theta\frac{\Delta}{\sigma^4}.
\end{aligned}
\end{equation}
The derivative $\partial_\phi\alpha^2$ is equal to zero. The metric does not depend on the coordinate $\phi$. The spatial derivatives of the non-zero component of $\beta^i$ are
\begin{equation}
\begin{aligned}
    \rightarrow \partial_r\beta^\phi &= -\frac{2Ma}{\sigma^2} + \frac{2Mra}{\sigma^4}\partial_r\sigma^2\\
    &= -\frac{2Ma}{\sigma^4}\Big[\big(\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta\big) - r\big(r^2\partial_r\rho^2 + 2r\rho^2 + a^2\partial_r\rho^2 + 2Ma^2\sin^2\theta\big)\Big]\\
    &= -\frac{2Ma}{\sigma^4}\Big[\big(r^4 + r^2a^2\cos^2\theta + r^2a^2 + a^4\cos^2\theta + \textcolor{blue}{2Mra^2\sin^2\theta}\big) - r\big(4r^3 + 2ra^2\cos^2\theta + 2ra^2 + \textcolor{blue}{2Ma^2\sin^2\theta}\big)\Big]\\
    &= -\frac{2Ma}{\sigma^4}\big[-3r^4 - r^2a^2\cos^2\theta - r^2a^2 + a^4\cos^2\theta\big]\\
    \rightarrow \partial_\theta\beta^\phi &= \frac{2Mra}{\sigma^4} \partial_\theta\sigma^2\\
    &= \frac{2Mra}{\sigma^4}\big[r^2\partial_\theta\rho^2 + a^2\partial_\theta\rho^2 + 4Mra^2\sin\theta\cos\theta\big]\\
    &= \frac{2Mra}{\sigma^4}\big[-2r^2a^2\sin\theta\cos\theta - 2a^4\sin\theta\cos\theta + 4Mra^2\sin\theta\cos\theta\big]\\
    &= -\frac{4Mra^3\Delta}{\sigma^4}\sin\theta\cos\theta\\
    \rightarrow \partial_\phi\beta^\phi &= 0
\end{aligned}
\end{equation}
The spatial derivatives of $\gamma^{rr}$ are
\begin{equation}
\begin{aligned}
    \rightarrow \partial_r\gamma^{rr} &= \frac{1}{\rho^4}\big[\rho^2\partial_r\Delta - \Delta\partial_r\rho^2\big]\\
    &= \frac{1}{\rho^4}\Big[\big(r^2 + a^2\cos^2\theta\big)(2r - 2M) - \big(r^2 - 2Mr + a^2\big)2r\Big]\\
    &= \frac{2}{\rho^4}\Big[ra^2\cos^2\theta + Mr^2 - Ma^2\cos^2\theta - ra^2\Big]\\
    &= \frac{2}{\rho^4}\Big[(r-M)a^2\cos^2\theta + (Mr-a^2)r\Big]\\
    \rightarrow \partial_\theta\gamma^{rr} &= -\frac{\Delta}{\rho^4}\partial_\theta\rho^2\\
    &= \frac{2a^2\Delta}{\rho^4}\sin\theta\cos\theta\\
    \rightarrow \partial_\phi\gamma^{rr} &= 0.
\end{aligned}
\end{equation}
The spatial derivatives of $\gamma^{\theta\theta}$ are
\begin{equation}
\begin{aligned}
    \rightarrow \partial_r\gamma^{\theta\theta} &= -\frac{1}{\rho^4}\partial_r\rho^2\\
    &= -\frac{2r}{\rho^4}\\
    \rightarrow \partial_\theta\gamma^{\theta\theta} &= -\frac{1}{\rho^4}\partial_\theta\rho^2\\
    &= \frac{2a^2}{\rho^4}\sin\theta\cos\theta\\
    \rightarrow \partial_\phi\gamma^{\theta\theta} &= 0.
\end{aligned}
\end{equation}
The spatial derivatives of $\gamma^{\phi\phi}$ are
\begin{equation}
\begin{aligned}
    \rightarrow \partial_r\gamma^{\phi\phi} &= \frac{1}{\sigma^4\sin^2\theta}\big[\sigma^2\partial_r\rho^2 - \rho^2\partial_r\sigma^2\big]\\
    &= \frac{1}{\sigma^4\sin^2\theta} \Big[2r\big(\rho^2r^2 + \rho^2a^2 + 2Mra^2\sin^2\theta\big) - \rho^2\big(4r^3 + 2ra^2\cos^2\theta + 2ra^2 + 2Ma^2\sin^2\theta\big)\Big]\\
    &= \frac{1}{\sigma^4\sin^2\theta} \Big[2r\big(r^4 + \textcolor{purple}{r^2a^2\cos^2\theta + r^2a^2} + \textcolor{blue}{a^4\cos^2\theta} + 2Mra^2\sin^2\theta\big)\\
    &\qquad\qquad\quad - r^2\big(4r^3 + \textcolor{purple}{2ra^2\cos^2\theta + 2ra^2} + 2Ma^2\sin^2\theta\big)\\
    &\qquad\qquad\quad - a^2\cos^2\theta\big(4r^3 + 2ra^2\cos^2\theta + \textcolor{blue}{2ra^2} + 2Ma^2\sin^2\theta\big)\Big]\\
    &= -\frac{2}{\sigma^4\sin^2\theta} \Big[r^2\big(r^3 - Ma^2\sin^2\theta\big) + a^2\cos^2\theta\big(2r^3 + ra^2\cos^2\theta + Ma^2\sin^2\theta\big)\Big]\\
    \rightarrow \partial_\theta\gamma^{\phi\phi} &= \frac{1}{\sigma^4\sin^3\theta}\big[\sigma^2\sin\theta\,\partial_\theta\rho^2 - \rho^2\sin\theta\,\partial_\theta\sigma^2 - 2\rho^2\sigma^2\cos\theta\big]\\
    &= \frac{\partial_\theta\alpha^2}{\Delta\sin^2\theta} - \frac{2\rho^2\cos\theta}{\sigma^2\sin^3\theta}\\
    &= -4Mra^2(r^2 + a^2)\frac{\cos\theta}{\sigma^4\sin\theta} - \frac{2\rho^2\cos\theta}{\sigma^2\sin^3\theta}\\
    &= -\frac{2\cos\theta}{\sigma^4\sin^3\theta}\Big[2Mra^2\sin^2\theta\,(r^2 + a^2) + \rho^2\sigma^2\Big]\\
    \rightarrow \partial_\phi\gamma^{\phi\phi} &= 0.
\end{aligned}
\end{equation}
To summarize, the components of the metric and their non-zero derivatives are:
\begin{equation}
\begin{aligned}
    &\alpha^2 = \frac{\rho^2\Delta}{\sigma^2},\qquad \beta^i = \begin{pmatrix}
    0\\
    0\\
    -\frac{2Mra}{\sigma^2}
    \end{pmatrix},\qquad \gamma^{ij} = \begin{pmatrix}
    \frac{\Delta}{\rho^2} & 0 & 0\\
    0 & \frac{1}{\rho^2} & 0\\
    0 & 0 & \frac{\rho^2}{\sigma^2\sin^2\theta}
    \end{pmatrix},\\
    &\partial_r\alpha^2 = \frac{2M}{\sigma^4}\Big[r^6 + r^4a^2(2\sin^2\theta + \cos^2\theta) - 4Mr^3a^2\sin^2\theta + r^2a^4(\sin^2\theta - \cos^2\theta) - a^6\cos^2\theta\Big],\\
    &\partial_\theta\alpha^2 = -4Mra^2(r^2 + a^2)\sin\theta\cos\theta\frac{\Delta}{\sigma^4},\\
    &\partial_r\beta^\phi = -\frac{2Ma}{\sigma^4}\big[-3r^4 - r^2a^2\cos^2\theta - r^2a^2 + a^4\cos^2\theta\big],\\
    &\partial_\theta\beta^\phi = -\frac{4Mra^3\Delta}{\sigma^4}\sin\theta\cos\theta,\\
    &\partial_r\gamma^{rr} = \frac{2}{\rho^4}\Big[(r-M)a^2\cos^2\theta + (Mr-a^2)r\Big],\hspace{30pt} \partial_\theta\gamma^{rr} = \frac{2a^2\Delta}{\rho^4}\sin\theta\cos\theta,\\
    &\partial_r\gamma^{\theta\theta} = -\frac{2r}{\rho^4},\hspace{165pt} \partial_\theta\gamma^{\theta\theta} = \frac{2a^2}{\rho^4}\sin\theta\cos\theta,\\
    &\partial_r\gamma^{\phi\phi} = -\frac{2}{\sigma^4\sin^2\theta} \Big[r^2\big(r^3 - Ma^2\sin^2\theta\big) + a^2\cos^2\theta\big(2r^3 + ra^2\cos^2\theta + Ma^2\sin^2\theta\big)\Big],\\
    &\partial_\theta\gamma^{\phi\phi} = -\frac{2\cos\theta}{\sigma^4\sin^3\theta}\Big[2Mra^2\sin^2\theta\,(r^2 + a^2) + \rho^2\sigma^2\Big].
\end{aligned}
\end{equation}


\subsection{The Kerr metric in Kerr-Schild coordinates}

The Kerr-Schild form of the Kerr metric is
\begin{equation}
    g_{\mu\nu} = \eta_{\mu\nu} + f k_\mu k_\nu \quad\text{with}\quad f\equiv\frac{2Mr^3}{r^4+a^2z^2} \quad\text{and}\quad k_\mu = \Big(1, \frac{rx+ay}{r^2+a^2}, \frac{ry-ax}{r^2+a^2}, \frac{z}{r}\Big)
\end{equation}
where $r$ is defined by the relation $\frac{x^2+y^2}{r^2+a^2} + \frac{z^2}{r^2} = 1$. In the limit that $a \to 0$, these coordinates become standard Cartesian coordinates. The formula for the parameter $r$ can be rewritten as
\begin{equation}
    r^2 = \frac{1}{2}\Big(x^2 + y^2 + z^2 - a^2 + \sqrt{\big(x^2 + y^2 + z^2 - a^2\big)^2 + 4a^2z^2}\,\Big)
\end{equation}
The spatial component $\gamma_{ij}$ and its inverse $\gamma^{ij}$ are
\begin{equation}
\begin{aligned}
    \gamma_{ij} &= \delta_{ij} + f k_i k_j\\
    \gamma^{ij} &= \frac{1}{\gamma} \text{adj}(\gamma_{ij}) \quad\text{(where we take the adjugate matrix)}\\
    &= \frac{1}{\gamma}\begin{pmatrix}
    (1+fk_2^2)(1+fk_3^2)-f^2k_2^2k_3^3 & -(1+fk_3^2)fk_1k_2+f^2k_1k_2k_3^2 & f^2k_1k_2^2k_3-(1+fk_2^2)fk_1k_3\\
    -(1+fk_3^2)fk_1k_2+f^2k_1k_2k_3^2 & (1+fk_1^2)(1+fk_3^2)-f^2k_1^2k_3^3 & -(1+fk_1^2)fk_2k_3+f^2k_1^2k_2k_3\\
    f^2k_1k_2^2k_3-(1+fk_2^2)fk_1k_3 & -(1+fk_1^2)fk_2k_3+f^2k_1^2k_2k_3 & (1+fk_1^2)(1+fk_2^2)-f^2k_1^2k_2^3
    \end{pmatrix}\\
    &= \frac{1}{\gamma}\begin{pmatrix}
    1+f(k_2^2+k_3^2) & -fk_1k_2 & -fk_1k_3\\
    -fk_1k_2 & 1+f(k_1^2+k_3^2) & -fk_2k_3\\
    -fk_1k_3 & -fk_2k_3 & 1+f(k_1^2+k_2^2)
    \end{pmatrix}\\
    &= \frac{1}{\gamma}\big[(1+fk_lk^l)\delta^{ij} - fk^ik^j\big].
\end{aligned}
\end{equation}
In the last line of this equation, we introduce $k^i$ with an upper index, of which the components are identical to $k_i$. More formally, we use $k^\mu \equiv \eta^{\mu\nu}k_\nu$. The determinant $\gamma$ can be computed as well:
\begin{equation}
\begin{aligned}
    \gamma &= \gamma_{11}(\gamma_{22}\gamma_{33}-\gamma_{23}\gamma_{32}) - \gamma_{12}(\gamma_{21}\gamma_{33}-\gamma_{23}\gamma_{31}) + \gamma_{13}(\gamma_{21}\gamma_{32}-\gamma_{22}\gamma_{32})\\
    &= (1+fk_1^2)\big(1+f(k_2^2+k_3^2)\big) + f^2k_1^2k_2^2 - f^2k_1^2k_3^2\\
    &= 1 + fk_ik^i.
\end{aligned}
\end{equation}
The components $\beta_i$ and $\beta^i$ of the metric are
\begin{equation}
\begin{aligned}
    \beta_i &= fk_i\\
    \beta^i &= f\gamma^{ij}k_j\\
    &= f\Big[\delta^{ij} - \frac{f}{1+fk_lk^l}k^ik^j\Big]k_j\\
    &= \frac{fk^i}{1+fk_lk^l}.
\end{aligned}
\end{equation}
The final component of the metric, $\alpha^2$, is
\begin{equation}
\begin{aligned}
    \alpha^2 &= -g_{00} + \beta_i\beta^i\\
    &= 1 - f + \frac{f^2k_ik^i}{1+fk_lk^l}\\
    &= 1 - \frac{f}{1+fk_lk^l}.
\end{aligned}
\end{equation}

\section*{ADM $3+1$ split}

\begin{align}
  \label{eq:ADM_line_elem}
  ds^2=-\alpha^2dt^2+(\beta_idt+dx_i)(\beta^idt+dx^i)
\end{align}
or
\begin{align}
  g_{\mu\nu}=\begin{pmatrix} -\alpha^2+\beta^2 & \beta_i \\ \beta_j & \gamma_{ij} \end{pmatrix}
\end{align}
Relation to general metric:
\begin{align}
  \label{eq:metric_to_ADM:al}
  \alpha^2&=\beta^2-g_{00}\\
  \beta_i&=g_{0i}\\
  \gamma_{ij}&=g_{ij}
\end{align}

Geodesics are given by
\begin{align}
  dx^i/dt&=\gamma^{ij}u_j/u^0-\beta^i\\
  du_i/dt&=-\alpha u^0\partial_i\alpha+u_k\partial_i\beta^k-u_ju_k(\partial_i\gamma^{jk})/(2u^0)
\end{align}
where
\begin{align}
  u^0=(\gamma^{jk}u_ju_k+\epsilon)^{1/2}/\alpha
\end{align}
with $\epsilon=0$ for lightlike and $\epsilon=1$ for timelike geodesics.

\section*{Kerr spacetime in Boyer-Lindquist coordinates}

\textit{This can almost trivially be extended to Boyer-Lindquist coordinates for a Kerr-Newman spacetime.}

Let $a=J/M$ be the Kerr parameter for angular momentum $J$ and mass $M$. Defining also the usual $\Sigma$ and $\Delta$, the Kerr metric in Boyer-Lindquist coordinates is given by
\begin{align}
  \Sigma&=r^2+a^2\cos^2\theta\\
  \Delta&=r(r-2M)+a^2\\
  \label{eq:Kerr_metric}
  ds^2&=-(dt-a\sin^2\theta d\phi)^2\Delta/\Sigma+(adt-(r^2+a^2)d\phi)^2\sin^2\theta/\Sigma+dr^2\Sigma/\Delta+\Sigma d\theta^2.
\end{align}
In the limit that $J,M\to0$, this reduces to
\begin{align}
  ds^2=-dt^2+dr^2+r^2d\Omega_2^2,
\end{align}
a flat spacetime with spatial slices in `spherical coordinates', this importantly gives $(r,\theta,\phi)$ their usual meaning.\\
Comparing \eqref{eq:Kerr_metric} to \eqref{eq:ADM_line_elem}, we can read off the $3{+}1$ split, and perform some simplifications.\footnote{One can implement the solver such that $g_{\mu\nu}$ is specified and compute the $3{+}1$ split numerically, but this may be numerically unstable due to catastrophic cancellation in \eqref{eq:metric_to_ADM:al}, who knows. Simplifying seems worthwhile.}
\begin{align}
  \gamma_{rr}&=\Sigma/\Delta\\
  \gamma_{\theta\theta}&=\Sigma\\
  \gamma_{\phi\phi}&=\big((r^2+a^2)^2-\Delta a^2\sin^2\theta\big)\sin^2\theta/\Sigma\\
  &=\sin^2\theta\big((r^2+a^2)+(2Ma^2r\sin^2\theta)/\Sigma\big)\\
  \gamma_{r\theta}&=\gamma_{r\phi}=\gamma_{\theta\phi}=0
\end{align}
\begin{align}
  \gamma^{rr}&=\Delta/\Sigma\\
  \gamma^{\theta\theta}&=1/\Sigma\\
  \gamma^{\phi\phi}&=1/\Big(\sin^2\theta\big((r^2+a^2)+(2Ma^2r\sin^2\theta)/\Sigma\big)\Big)\\
  \gamma^{r\theta}&=\gamma^{r\phi}=\gamma^{\theta\phi}=0
\end{align}
\begin{align}
  \beta_r&=\beta_\theta=0\\
  2\beta_\phi&=2a\sin^2\theta\,\Delta/\Sigma-2a(r^2+a^2)\sin^2\theta/\Sigma\\
  &=2a\sin^2\theta\,(\Delta-r^2-a^2)/\Sigma\\
  \beta_\phi&=-2Mar\sin^2\theta/\Sigma
\end{align}
\begin{align}
  \beta^r&=\beta^\theta=0\\
  \beta^\phi&=\gamma^{\phi i}\beta_i=\gamma^{\phi\phi}\beta_\phi\\
  &=\frac{-2Mar}{(r^2+a^2)\Sigma+2Ma^2r\sin^2\theta}
\end{align}
\begin{align}
  \beta^2&=\beta_i\beta^i=\beta_\phi\beta^\phi\\
  &=\frac{4M^2a^2r^2\sin^2\theta}{(r^2+a^2)\Sigma^2+2Ma^2r\Sigma\sin^2\theta}
\end{align}
\begin{equation}
\begin{split}
  -\alpha^2+\beta^2&=-\Delta/\Sigma+a^2\sin^2\theta/\Sigma\\
  \alpha^2&=\beta^2+\Delta/\Sigma-a^2\sin^2\theta/\Sigma\\
  &=\frac{4M^2a^2r^2\sin^2\theta}{(r^2+a^2)\Sigma^2+2Ma^2r\Sigma\sin^2\theta}+\frac{\Sigma-2Mr}\Sigma\\
  &=\frac{(r^2+a^2)\Sigma+2Ma^2r\sin^2\theta-2Mr(r^2+a^2)}{(r^2+a^2)\Sigma+2Ma^2r\sin^2\theta}\\
  &=\frac{(r^2+a^2)\Sigma-2Mr^3-2Mra^2\cos^2\theta}{(r^2+a^2)\Sigma+2Ma^2r\sin^2\theta}\\
  &=\frac{(r^2+a^2)\Sigma-2Mr\Sigma}{(r^2+a^2)\Sigma+2Ma^2r\sin^2\theta}\\
  &=\frac{\Delta\Sigma}{(r^2+a^2)\Sigma+2Ma^2r\sin^2\theta}
\end{split}
\end{equation}

\section*{Particle Trajectories in Curved Spacetime}
In this section, we will discuss the qualitative nature of the trajectories of both massless and massive particles in the vicinity of black holes. The case under consideration will be that of the Kerr Black Hole, which can be trivially extended to the Schwarzschild solution as well. Our analysis will only cover particles in the region outside the outermost event horizon.

\subsection*{Kerr Spacetime}
First, we once again specify the Kerr metric in the standard Boyer-Lindquist $(t,r,\theta,\phi)$ coordinates. We have taken the gravitational constant and the speed of light both to be unity,

\begin{equation}
    ds^2 = -(1-\frac{2Mr}{\Sigma})dt^2 - \frac{4Mr}{\Sigma}a \sin^2\theta dtd\phi + \Sigma(\frac{dr^2}{\Delta} + d\theta^2) + (r^2 + a^2 + \frac{2Mr}{\Sigma}a^2\sin^2\theta)\sin^2\theta d\phi^2
\end{equation}

where,

\[
    \Sigma = r^2 + a^2\cos^2\theta, \qquad \Delta = r^2 - 2Mr + a^2
\]

and $M$ and $a$ represent the mass and the angular momentum per unit mass of the black hole respectively.\medskip

There are three interesting radial positions to note - two of these are the event horizons that are derived from the roots of $\Delta$ which make the metric singular - ($r_-$ and $r_+$).

The other is the killing horizon where the time Killing vector becomes spacelike - which is the boundary of the ergosphere.
\[
    r_{\pm} = M \pm \sqrt{M^2 - a^2}, \qquad r_{ergo} = M + \sqrt{M^2 - a^2\cos^2\theta}
\]
\\
We will also briefly mention the nomenclature that will be used to classify the trajectories in this report. Namely,

\begin{itemize}
    \item \parHeading{Lightlike vs Timelike}: Lightlike trajectories are those followed by massless particles such as photons, while timelike trajectories are those followed by massive particles. The main difference between the two is that lightlike trajectories have a four-velocity that is null, while timelike trajectories have a four-velocity that is timelike.
    \item \parHeading{Spherical (Circular) vs Non-spherical (Non-circular)}: Spherical trajectories are those that maintain a constant radial coordinate, while non-spherical trajectories are those that do not. When considering only the equatorial plane, these are referred to as circular and non-circular trajectories respectively.
    \item \parHeading{Equatorial vs Non-equatorial}: Equatorial trajectories are those that lie only in the equatorial plane - $\theta = \pi/2$ - while, non-equatorial trajectories are those that do not.
\end{itemize}

Corresponding to this metric, the geodesic equations parameterized by the proper time ($\tau$) are as follows. We have also defined a corollary for the $\theta$ coordinate as $u = \cos\theta$, which represents the latitudinal position of the particle with respect to the black hole.

\begin{subequations} \label{eq:geodesic_equations}
\begin{align}
   \Sigma\frac{dr}{d\tau} = \pm\sqrt{R(r)} \\
   \Sigma\frac{du}{d\tau} = \pm\sqrt{V(u)} \\ 
   \Sigma\frac{d\phi}{d\tau} = \frac{\Phi}{1-u^2} + \frac{a}{\Delta}(2MrE - a\phi) \\
   \Sigma\frac{dt}{d\tau} = -a^2E(1-u^2) + \frac{1}{\Delta}(E(r^2 + a^2)^2 - 2Mra\Phi)
\end{align}
\end{subequations}

where,

\begin{equation} \label{eq:radial_potential}
    R(r) = (E^2 - \mu^2)r^4 + 2M\mu^2r^3 + [a^2(E^2 - \mu^2) - Q - \Phi^2]r^2 + 2M[(aE - \Phi)^2 + Q]r - a^2Q
\end{equation}

\begin{equation} \label{eq:theta_potential}
    V(u) = a^2(\mu^2 - E^2)u^4 - [Q + \Phi^2 + a^2(\mu^2 - E^2)]u^2 + Q
\end{equation}

Here, $\mu$ is the rest mass of the particle. In the case of massive particles we take it to be equal to 1 and in the case of massless particles it is equal to 0. $E$ and $\Phi$ are the energy and the angular momentum of the particle in the direction of rotation of the black hole and $Q$ is the Carter's constant. We will elaborate on the nature of the Carter's constant in the next section, but for now we will just treat it as a constant of motion that is related to the motion of the particle in the $\theta$ direction.

The right hand side of (\ref{eq:geodesic_equations}) represents the total amount of kinetic energy available to the particle - its total energy minus the effective potential it has to overcome.\medskip

We can gain some information about this effective potential by looking at the radial geodesic equation for a specific class of trajectories,
\begin{itemize}
    \item We consider a massless photon moving on null/lightlike geodesics.
    \item The photon starts its motion in the equatorial plane ($\theta = \pi/2$) with zero Carter's Constant.
    \item It is either co-rotating or counter-rotating with respect to the black hole, so $\Phi$ is non-zero.
\end{itemize}

\parHeading{Non-Spherical, Lightlike, Equatorial Motion:} When the RHS of the radial and latitudinal equations ($R(r)$ and $V(u)$) are zero, it represents a turning point where the effective potential is equal to the energy of the particle - causing it to stop. If there is a non-zero force acting on the particle at this point (first derivative of the RHS functions) then it will reverse its motion in that particular direction. \medskip

For the radial case with the conditions that we mentioned above, the RHS reduces to,
\begin{equation}
    \dot{r}^2 = \frac{(r^2+a^2)^2-a^2\Delta}{r^4}(E-V_-)(E-V_+) + \frac{\kappa\Delta}{r^2}
\end{equation}

where,
\begin{equation}
   \boxed{V_{\pm} = \frac{2Mar\Phi \pm r^2|\Phi|\sqrt{\Delta}}{(r^2+a^2)^2-a^2\Delta}}
\end{equation}

$\kappa$ is the constant that leads to the parametrization of the four velocity. In the photon case $\kappa = 0$ since we are considering massless particles.\medskip

The absolute value of $\Phi$ is taken to make sure that $V_+$ is always greater than $V_-$ in the case that the particle is counter-rotating or co-rotating with respect to the black hole.\bigskip

We now state some properties of the effective potentials,

\begin{itemize}
    \item Both the potentials coincide at the outer horizon when $\Delta=0$. This converged potential is positive for co-rotating and negative for counter-rotating photons. Since, we only consider motion outside the outer event horizon for now, therefore, our area of interest starts here. 
    \item At infinity, both the potentials go to zero and they only have one extremum each - which is always a maximum for $V_+$ and a minimum for $V_-$. 
    \item Owing to the presence of the ergosphere, the object can have negative energies in specific regions as we will see. 
\end{itemize}

For allowed ranges of motion the RHS should be non-negative, which can only happen at the following ranges. 

\[ E < V_-, \qquad E > V_+ \]

Any motion outside these two ranges is forbidden since the particle does not have enough energy to overcome the effective potential. Figures \ref{fig:counter_potential_sketch} and \ref{fig:co_potential_sketch} are sketches of the counter-rotating and co-rotating potentials for a specific value of $\Phi$, $M$ and $a$.

\begin{figure}[H]     
    \centering
    \includegraphics[width=0.83\textwidth]{figures/kerr_potential_counterrotating.png}
    \caption{Sketches of Effective Potential for Counter-Rotating Photons in Kerr Spacetime.}
    \label{fig:counter_potential_sketch}
\end{figure}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.83\textwidth]{figures/kerr_potential_corrotating.png}
    \caption{Sketches of Effective Potential for Co-Rotating Photons in Kerr Spacetime.}
    \label{fig:co_potential_sketch}
\end{figure}

One case during this motion is when the particle starts from far away with some finite energy and approaches the black hole. It will be able to cross the horizon if its energy is greater than the maximum of $V_+$. This is a case of photon capture.\smallskip

If it is less than the maximum of $V_+$ then it will have a turning point somewhere outside the horizon and will reverse its direction back to infinity. This is the case of deflection.\bigskip

The other case pertains to motion of co-rotating photons between the outer event horizon and the maximum of $V_+$. It is only allowed to exist in this region within the range of $V_+(r_+) < E < V_+(r_{max})$.\smallskip

The last case is that of counter-rotating photons between the ergosphere (where $V_+ = 0$) and the outer event horizon. Here, the energy of the particle can be negative and it can exist in the range of $V_+(r_+) < E < 0$. The existence of this "negative energy" is due to the fact that the time Killing vector becomes spacelike in the ergosphere and therefore, the energy of the particle as measured by an observer can be negative. This is the same concept used during the description of the Penrose process. \bigskip

We can also see that there are two unstable circular orbits present here - one prograde (co-rotating case) and one retrograde (counter-rotating case). These orbits are located at the maximum of $V_+$ and will be discussed in detail below in the section on spherical orbits. \bigskip

\begin{framed}
\textbf{Note:} Finally, we can expand this discussion to all non-spherical trajectories - including those that are timelike and lie outside the equatorial plane. Their qualitative nature also follows the same treatment. Although, in such cases, the energy ranges will get modified due to the presence of the non-zero $\kappa$ and $\theta$ terms in the geodesic equations. 
\end{framed}

\subsection*{Spherical Orbits}
Spherical orbits constitute those special cases where the radial coordinate remains constant throughout the motion of the particle. However, the particle is free to move in the $\theta$ and $\phi$ directions. 

The conditions for spherical orbits can be derived from the radial geodesic equation (\ref{eq:geodesic_equations}a). For a spherical orbit, the radial velocity and the radial acceleration must both be zero. This leads to the following conditions:

\[ R(r) =0, \qquad \dot{R}(r) =0 \]

There are, however, restrictions on the motion of the particle in the $\theta$ plane. $V(u)$ in the $\theta$ geodesic equation (\ref{eq:geodesic_equations}b) must also be non-negative for the particle to be able to move in the latitudinal direction. This gives us the following condition,

\[ V(u) \geq 0, \qquad |u| < 1\]

Therefore, we have to look for values of u for which the above conditions are satisfied - which is done by looking at the behavior of the roots of $V(u)$. The solutions of this equation allow for the latitude variable, $u$, to exist within a certain range. At the boundaries of this range, the particle will have a turning point in the $\theta$ direction and will reverse its motion.\smallskip 

Therefore, the motion in the $\theta$ direction is oscillatory between these two turning points.\medskip

Now it is time to analyze this periodic motion in the context of timelike and lightlike geodesics.

\subsubsection*{Lightlike Trajectories}
We can rewrite the $\theta$ geodesic equation (\ref{eq:geodesic_equations}b) for photons as follows,

\begin{equation} \label{eq:theta_geodesic_photon}
    (\frac{\Sigma}{E})^2 \dot{u}^2 = \mathcal{Q} - (\mathcal{Q} + \varPhi^2 -a^2)u^2 - a^2u^4
\end{equation}

where,

\[ \mathcal{Q} = \frac{Q}{E^2}, \qquad \varPhi = \frac{\Phi}{E} \]

This equation is quadratic in $w = u^2$ and gives us an idea about the roots of $V(u)$ depending on the sign of $\mathcal{Q}$.

According to the conditions outlined in the previous section, non-negative values of $w$ can only exist when $\mathcal{Q}\geq 0$ \cite{taoPhotonSpherical}. Therefore, we will consider only this case for the rest of our discussion. The positive roots for the above equation for non-negative $\mathcal{Q}$ are given by,

\begin{subequations}
\begin{align*}
    u_0^2 &= \frac{(a^2 - \mathcal{Q} - \varPhi^2) + \sqrt{(a^2 - \mathcal{Q} - \varPhi^2)^2 + 4a^2\mathcal{Q}}}{2a^2} & {\mathcal{Q}>0} \\
    u_0^2 &= 0 & {\mathcal{Q} = 0} \\
\end{align*}
\end{subequations}

\parHeading{$\mathcal{Q}>0$}: Here the particle reaches a maximum latitude of $u_0$ and then reverses its direction. The motion in the $\theta$ direction is oscillatory between $-u_0$ and $u_0$.\medskip

\parHeading{$\mathcal{Q}=0$}: This represents an unstable circular orbit where the particle starts in the equatorial plane with zero velocity in the $\theta$ direction and stays in the equatorial plane. These are the unstable circular orbits described above.

\begin{framed}
    \parHeading{Note:} If we set $u=0$ (equatorial plane) in the geodesic equation (\ref{eq:theta_geodesic_photon}) we get $r^2\dot{\theta} = \pm\sqrt\mathcal{Q}$ which is the condition for equatorial orbits. Therefore, physically, the Carter's constant can be interpreted as a measure of the angular speed of the photon while crossing the Black Hole's equator. 
\end{framed}

Now that we have deduced the motion in the $\theta$ direction, we can turn to the radial geodesic equation. Setting $R(r)$ and its derivative to zero, we can find the conditions for spherical photon orbits. These restrictions give us the expressions for $\mathcal{Q}$ and $\varPhi$ in terms of the radius of the orbit, $r$,

\begin{equation} \label{eq:phi_spherical_photon}
    \varPhi = -\frac{r^3 - 3Mr^2 + a^2r + a^2M}{a(r-M)}
\end{equation}

\begin{equation} \label{eq:Q_spherical_photon}
    \mathcal{Q} = -\frac{r^3(r^3 - 6Mr^2 + 9M^2r - 4a^2M)}{a^2(r-M)^2}
\end{equation}

The azimuthal direction requires a more careful analysis. The $\phi$ geodesic equation can be rewritten in terms of $w$ using (\ref{eq:geodesic_equations}c) and (\ref{eq:geodesic_equations}d) as follows,

\begin{equation}
    \frac{d\phi}{dw} = \frac{1}{Y(w)}\Big[\frac{\varPhi}{2(1-w)} + \frac{2Mra - a^2\phi}{2\Delta}\Big]
\end{equation}
where
\[ Y^2(w) = \mathcal{Q}w - (\mathcal{Q} + \varPhi^2 -a^2)w^2 - a^2w^3 \]

Integrating these expressions can provide us with analytical values of $\Delta\phi$ for one period of oscillation in the $\theta$ direction. 

The velocity in the $\phi$ direction is not always prograde or retrograde. From (\ref{eq:geodesic_equations}c), we can see that $\dot{\phi}$ changes sign when $u$ hits the following value,

\begin{equation}
    u_1^2 = \frac{r^2(3M-r)}{a^2(r+M)}
\end{equation}

Of course, this is only relevant when the photon is actually able to reach this point or when $u_1 < u_0$. 

\begin{framed}
Therefore, the velocity in the $\phi$ direction may reverse at certain points. But, overall, the orbit can be described as retrograde or prograde depending on the sign of $\varPhi$. As it is representative of the total displacement in the $\phi$ direction per latitudinal oscillation. 
\end{framed}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.9\textwidth]{figures/kerr_photon.png}
    \caption{Variation of $\varPhi, \mathcal{Q}, u_0$ and $u_1$ with changing radius of the orbit. Note that these values are only considered in the region between the two circular orbits as discussed.}
    \label{fig:parameter_values_photon}
\end{figure}

In the plot (Figure \ref{fig:parameter_values_photon}), we see that $\mathcal{Q}$ is positive in the range between two specific radii. These two radii represent the two unstable circular orbits in the equatorial plane - one is the prograde and the other is the retrograde one. The radii of these orbits are,

\begin{subequations}
    \begin{align}
        r_1 &= 2M\Big\{1 + \cos\Big[\frac{2}{3}\arccos\Big(-\frac{|a|}{M}\Big)\Big]\Big\} \\
        r_2 &= 2M\Big\{1 + \cos\Big[\frac{2}{3}\arccos\Big(\frac{|a|}{M}\Big)\Big]\Big\}
    \end{align}
\end{subequations}

We also see that the increase in the maximum latitude of the photon is associated with a decrease in the angular momentum of the photon in the $\phi$ direction. This relates to the conservation of total angular momentum for the photon. $\varPhi$ hits zero at the same point when $u_0$ hits its maximum value of $1$ where it can reach the poles. The spherical orbits shift from being prograde to retrograde at the radius where $\varPhi$ changes sign, given by,

\begin{equation}
    r_3 = M + 2\sqrt{M^2 - \frac{a^2}{3}}\cos\Big[\frac{1}{3}\arccos\frac{M(M^2-a^2)}{(M^2 - \frac{a^2}{3})^{\frac{3}{2}}}\Big]
\end{equation}

\parHeading{Note:} A word about $\Delta \phi$ - it has a discontinuity at $r_3$. This occurs because the photon reaches its maximum latitude at the poles during this orbit. A small perturbation on either side would add or subtract a full $2\pi$ to the total displacement in the $\phi$ direction. In one case it would swing around the poles - adding a $2\pi$ - and in the other it would miss the poles and end up with a retrograde displacement - subtracting a $2\pi$.

\subsubsection*{Timelike Trajectories}
The situation is more complicated in the case of timelike trajectories as the non-zero $\mu$ does not allow us to effectively factor out the energy as a normalization factor - as we did in the case of photons. Here the roots of $V(u)$ are as follows for the case of $\mu = 1$,

\begin{equation} \label{eq:w_roots}
   w_{1,2} = u_{1,2}^2 = \frac{1}{2a^2(1-E^2)}\Big[a^2(1-E^2) + \Phi^2 + Q \mp \sqrt{(a^2(1-E^2) + \Phi^2 + Q)^2 - 4a^2Q(1-E^2)}\Big]
\end{equation}

Similarly, the radial geodesic equation cannot be parametrized in terms of $\mathcal{Q}$ and $\varPhi$ and will be better off being parametrized by $r$ and $Q$. Therefore, using the velocity and acceleration conditions for spherical orbits, the energy and angular momentum of the photon can be written as,

\begin{equation} \label{eq:energy_spherical_timelike}
    E_{a,b} = \frac{r^3(r-2M) - a(aQ \mp \sqrt{\Upsilon})}{r^2\sqrt{r^3(r-3M) -2a(aQ \mp \sqrt{\Upsilon})}}
\end{equation}

\begin{equation} \label{eq:angular_momentum_spherical_timelike}
    \Phi_{a,b} = -\frac{2Mar^3 + (r^2+a^2)(aQ \mp \sqrt{\Upsilon})}{r^2\sqrt{r^3(r-3M) -2a(aQ \mp \sqrt{\Upsilon})}}
\end{equation}

where,

\begin{equation}
    \Upsilon = Mr^5 - Q(r-3M)r^3 + a^2Q^2
\end{equation}

There are actually four branches of the solutions here. The above $(a,b)$ branches are those with positive energies and the other two - $(c,d)$ - are those with negative energies. Negative energies can exist in the ergosphere of the Black Hole. We will analyze the positive energy solutions in this paper and the negative cases can be trivially obtained by $E_{c,d} = -E_{a,b}$.

Using the restrictions on the various square roots and the non-negative nature of both $R(r)$ and $V(u)$, we can define the parameter space (values of $r$ and $Q$) for which spherical orbits can exist. 

Before we produce the parameter space, we will discuss the nature of these orbits. Namely, the orbits will be classified on the basis of the following criteria:

\begin{itemize}
    \item \parHeading{Prograde vs Retrograde}: Prograde orbits are those that are co-rotating with the black hole, while retrograde orbits are those that are counter-rotating. The sign of $\Phi$ can be used to determine whether the orbit is prograde or retrograde.
    \item \parHeading{Stable vs Unstable}: Stable orbits are those that are stable under small perturbations, while unstable orbits are those that are not. The stability of an orbit can be determined by analyzing the second derivative of $R(r)$ (\ref{eq:radial_potential}) at the radius of the orbit.
    \item \parHeading{Bound vs Unbound}: Bound orbits are those that permit values in a finite range of radii outside the event horizon, while unbound orbits extend till infinity. The criteria to distinguish them is given by the behavior of $R(r)$ at infinity. At large radii, only the leading coefficient of $(E^2 - \mu^2)$ plays a role in determining the sign of $R(r)$. If $R(r)$ is positive until infinity, then the orbit is unbound as motion even at diverging radii is allowed. If $R(r)$ starts to tend to negative values at $r$ increases then it is a bound orbit and the particle cannot escape to infinity. These are respectively the cases of $E^2 > 1$ and $E^2 < 1$ for timelike trajectories - with $E^2 = 1$ being the critical case of marginally bound orbits.
\end{itemize}

All three of these criteria can be determined by looking at the value of $Q$. Namely, boundaries of $Q$ exist for each radius $r$ that separate the parameter space into different regions corresponding to different orbits. The marginal values of $Q$ that define these boundaries are given by $Q_{mb}, Q_{ms}, Q_o$ for the bound, stable and prograde/retrograde boundaries respectively. These are given by the following expressions,

\begin{subequations}
    \begin{align}
        Q_{ms} = -\frac{Mr^{5/2}[(\sqrt{\Delta} - 2\sqrt{Mr})^2 - 4a^2]}{4a^2(r^{3/2} - M\sqrt{r}- \sqrt{M \Delta})} \\
        Q_{mb} = -\frac{Mr^2[r(\sqrt{r} - \sqrt{2M})^2 - 4a^2]}{a^2(\sqrt{r} - \sqrt{M})^2} \\
        Q_o = \frac{Mr^2(\Delta^2 + 4Mr^2(r-M))}{(r^2+a^2)(r\Delta - M(r^2-a^2))}
    \end{align}
\end{subequations}

Keep in mind that the criterion related to angular momentum and $Q_o$ is only relevant past a certain radius, as $Q_o$ is negative for radii smaller than a certain $r_o$. This radius is given by the following expression and the orbits are all prograde for $r<r_o$ irrespective of the value of $Q$ in this range.

\[ r_o = M + 2\sqrt{\frac{3M^2-a^2}{3}}\cos\Big(\frac{1}{3}\arccos\Big(\frac{3M(M^2-a^2)}{3M^2 - a^2}\sqrt{\frac{3}{3M^2-a^2}}\Big)\Big) \]\bigskip

\parHeading{Summary of Timelike Spherical Orbits:}
The various regimes of solutions can be divided using the two photon circular orbits of radii $r_1$ and $r_2$,
\begin{itemize}
    \item \parHeading{$r = r_1, a=M$}: For the extremal Kerr black hole, there is a stable circular prograde orbit at the horizon. This is a special case and is called a horizon-skimming orbit. For more information about this case, see \cite{taoTimelikeSpherical}.
    \item \parHeading{$r_1 \leq r < r_2$}: There is always a $Q=0$ solution that corresponds to a stable circular prograde orbit. The particle starts its latitudinal oscillations (much like the photons) when $Q$ is increased. In this range, only the first class ($a$) of solutions of energy (\ref{eq:energy_spherical_timelike}) and angular momentum (\ref{eq:angular_momentum_spherical_timelike}) are possible and $Q$ can take any positive value. As $Q$ is increased, the orbits become to begin to move towards the unstable and unbound range. 
    \item \parHeading{$r = r_2$}: Here, there are restrictions on the value of $Q$ and it can only follow the range $ 0 \leq Q < 32M^2$. As $Q$ moves towards the upper limit, this orbit reduces to the unstable circular retrograde orbit at this radius. Only the $a$ branch of the solutions exist here.
    \item \parHeading{$r>r_2$}: The range of $Q$ is equal to $0 \leq Q < Q_1$ where the upper limit has the following expression,
    \[Q_1 = \frac{r^2}{2a^2}(r(r-3M)\mp \sqrt{r\Xi})\]
    where,
    \[\Xi = r^3 - 6Mr^2 + 9M^2r - 4Ma^2\]
    \begin{framed}
    The $32M^2$ upper-limit in the previous case is the same that will be derived from this expression when $r=r_2$.
    \end{framed}
    However, now both branches $(a,b)$ are feasible with different parameter spaces. These two solutions converge at the the upper limit of $Q=Q_1$ and all orbits in the second branch are retrograde. 
\end{itemize}

Finally, we end our discussion with certain analytical solutions of the geodesical equation that can be found using Mino's Method \cite{taoTimelikeSpherical}. 

We will not go into the details of the derivation but the basis is the shift from using proper time as a parameter to, $\lambda$ which is given by,

\[ \Sigma \frac{x^\mu}{d\tau} = \frac{dx^\mu}{d\lambda}\]

and called Mino's parameter.\medskip

Here, we can write the geodesic equation in the $\theta$ direction as follows,

\begin{equation}
    \frac{dw}{d\lambda} = 2Y(w)
\end{equation}

where,

\[ Y^2(w) = a^2(1-E^2)w(w-w_1)(w-w_2) \]

where $w_{1,2}$ are the given by (\ref{eq:w_roots}).\medskip

Now, we integrate the equation - starting from the equator- to get $\lambda$ as a function of $w$, which can then be inverted.

\begin{equation}
    \lambda(w) = \int_0^{w} \frac{dw}{2Y(w)} = \frac{1}{a\sqrt{(1-E^2)w_2}F(\psi,k)}
\end{equation}
where,
\[ \psi = \arcsin\sqrt{\frac{w}{w_1}}, \quad k = \sqrt{\frac{w_1}{w_2}} \]
Here, $F(\psi,k)$ is the incomplete elliptic integral of the first kind. 

Inverting this, we get the following expression for $\psi$,

\begin{equation}
    \psi = \operatorname{am}(a\sqrt{(1-E^2)w_2}\lambda, k)
\end{equation}

A full oscillation continues from $\lambda = 0$ to $4\lambda_0$ where,

\[ \lambda_0 = \frac{K(k)}{a\sqrt{(1-E^2)w_2}} \]

Similarly, the $\phi$ and $t$ coordinates can also be expressed in terms of $w$ by integrating the geodesic equation (\ref{eq:geodesic_equations}). We get,

\begin{subequations}
    \begin{align}
    \phi(\lambda) = \frac{\Phi}{a\sqrt{(1-E^2)w_2}}\Pi(\psi, w_1, k) + (2MrE - a\Phi)\frac{a}{\Delta} \\
    t(\lambda) = -\frac{aE}{a\sqrt{(1-E^2)w_2}}\Big[ (1-w_2)F(\psi,k) + w_2E(\psi,k) \Big] + \frac{1}{\Delta}\Big[ E(r^2 + a^2)^2 - 2Mra\Phi \Big]\lambda
    \end{align}
\end{subequations}

Where $E(\psi,k)$ and $\Pi(\psi, w_1, k)$ are the incomplete elliptic integral of the second kind and third kind. While, $K(k)$ is the complete elliptic integral of the first kind.

These integrals will only be consistent with the periodicity of the $\theta$ and $\phi$ motion if the following conditions are satisfied,

\begin{subequations}
    \begin{align}
        F(-\psi, k) = - F(\psi, k) && F(\psi+ \pi, k) = F(\psi, k) + 2K(k)\\
        \Pi(-\psi, w_1, k) = - \Pi(\psi, w_1, k) && \Pi(\psi + \pi, w_1, k) = \Pi(\psi, w_1, k) + 2\Pi(w_1, k)\\
        E(-\psi, k) = - E(\psi, k) && E(\psi + \pi, k) = E(\psi, k) + 2E(k)
    \end{align}
\end{subequations}

Keep in mind that this solution is valid only for bound orbits or $E^2<1$. For unbound orbits, the solution changes as follows,

\begin{subequations}
    \begin{align}
        k &= \sqrt{\frac{w_1}{w_1 - w_2}}, \qquad k \in (0,1) \\
        \psi &= \arcsin\sqrt{\frac{w(w_1 - w_2)}{w_1(w-w_2)}} = \operatorname{am}(a\sqrt{(E^2-1)(w_1-w_2)}\,\lambda,\, k)
    \end{align}
\end{subequations}

\begin{subequations}
    \begin{align}
        \phi(\lambda) &= \frac{\Phi}{a(1-w_2)\sqrt{(E^2-1)(w_1-w_2)}}\Big[F(\psi, k) - w_2\,\Pi(\psi,\, k^2(1-w_2),\, k)\Big]
        + \frac{a}{\Delta}(2MrE - a\Phi)\,\lambda \\
        t(\lambda) &= -\frac{aE}{\sqrt{(E^2-1)(w_1-w_2)}}\Big[ (1-w_2)F(\psi,k) + w_2\,\Pi\!\left(\psi,\, k^2,\, k\right)\Big]
        + \frac{1}{\Delta}\Big[ E(r^2+a^2)^2 - 2Mra\Phi \Big]\lambda
    \end{align}
\end{subequations}

\begin{equation}
    \lambda_0 = \frac{K(k)}{a\sqrt{(E^2-1)(w_1-w_2)}}
\end{equation}
